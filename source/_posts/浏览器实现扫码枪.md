# 浏览器实现扫码枪

先上代码，整个依赖了 quagga 这个库进行前端的一纬码解码 通过比较每次扫码出现的值一样出现一个阀值来判断是否扫到正确的数字

```vue
<template>
  <div v-loading="loading" element-loading-text="请稍后" class="scanner">
    <!-- 扫码完成后播放音频 -->
    <audio src="barcode-scan.mp3" style="display:none" ref="sound"></audio>
    <div
      class="scanner-container"
      ref="container"
      style="width: 100%;height: 100%;"
    />
    <!-- 扫描器模拟 -->
    <div v-if="!loading && !errorMessage" class="scanner-animation">
      <div class="diode">
        <div class="laser" />
      </div>
    </div>
    <div v-if="errorMessage">{{ errorMessage }}</div>
  </div>
</template>

<script>
/** @format */
import Quagga from "quagga";
import { Message } from "element-ui";
import debounce from "lodash/debounce";
const MAX_TIMES = 5; // 最大同步次数，超过这个值说明是正确的，可以进行核销

/**
 * 简单来说就是一个全屏扫码页，页面扫中某一个code超过规定次就当扫描成功，回调
 */
export default {
  data() {
    return {
      loading: false,
      barcode: "",
      aimTimes: 0,
      errorMessage: "",
      delay: 0
    };
  },
  methods: {
    onDetected({ codeResult }) {
      const { code } = codeResult;
      console.log("找到", code);
      if (code !== this.barcode) {
        this.barcode = code;
        this.aimTimes = 0;
      } else {
        this.aimTimes++;
      }
    },
    startScan: debounce(function(delay = 0) {
      setTimeout(() => {
        this.delay = delay;
        this.aimTimes = 0;
        this.barcode = "";
        this.loading = false;
        Quagga.start();
      }, delay);
    }, 500),
    stopScan() {
      this.aimTimes = 0;
      this.loading = true;
      Quagga.pause();
    }
  },
  watch: {
    aimTimes(times) {
      /**
       * 监听是否命中一定次数，如果命中了，发送并且锁上
       */
      if (times > MAX_TIMES) {
        this.stopScan();
        this.$refs.sound.play();
        this.$emit("input", this.barcode);
      }
    }
  },
  mounted() {
    let that = this;
    /**
     * mounted初始化
     */
    that.loading = true;
    Quagga.init(
      {
        inputStream: {
          name: "Live",
          type: "LiveStream",
          target: this.$refs.container
        },
        decoder: {
          readers: ["i2of5_reader", "2of5_reader"]
        }
      },
      function(err) {
        that.loading = false;
        if (err) {
          console.log(err.message);
          if (!err.message.includes("Cannot read property")) {
            that.errorMessage = err.message || err;
            Message({
              message: that.errorMessage,
              type: "error"
            });
          }
          return;
        }

        Quagga.onDetected(that.onDetected.bind(that));
        that.startScan();
      }
    );
  },
  beforeDestroy() {
    Quagga.stop();
    Quagga.offDetected(this.onDetected);
  }
};
</script>

}

  }

    height: 100%;
    height: 50%;

    width: 100%;
    position: absolute;
    top: 0;
    bottom: 0;
    margin: auto;
    .diode {
      animation: beam 0.01s infinite;
      .laser {
        width: 90%;
        margin-left: 5%;
        background-color: rgb(255, 99, 71);
        height: 1px;
        position: absolute;
        top: 0;
        z-index: 2;
        box-shadow: 0 0 4px #ff0000;
      }
    }
  }
}
</style>
```
